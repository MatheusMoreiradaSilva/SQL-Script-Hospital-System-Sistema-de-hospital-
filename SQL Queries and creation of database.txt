-- Creates the database
CREATE DATABASE Hospital

USE Hospital
GO

-- Creates the medical specialties table
CREATE TABLE Tb_Specialties(
    ID INT PRIMARY KEY NOT NULL,
    NAME VARCHAR(50) NOT NULL)


-- Creates the doctors table
CREATE TABLE Tb_Doctors(
    ID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    NAME VARCHAR(50) NOT NULL,
    LICENSE VARCHAR(20) NOT NULL,
    SPECIALTY_ID INT NULL,

    -- Links the doctor table with the specialty ID
    CONSTRAINT FK_Specialty_Doctor
        FOREIGN KEY (SPECIALTY_ID) REFERENCES Tb_Specialties (ID)
)


-- Creates the patients table
CREATE TABLE Tb_Patients(
    ID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    NAME VARCHAR(50) NOT NULL,
    DATE_OF_BIRTH DATE NULL,
    PHONE VARCHAR(20) NULL)


-- Creates the appointments table
CREATE TABLE Tb_Appointments(
    ID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    PATIENT_ID INT NOT NULL,
    DOCTOR_ID INT NOT NULL,
    APPOINTMENT_DATE DATE NOT NULL,
    DIAGNOSIS VARCHAR(100) NULL,

    /* Links the doctors and patients tables using both IDs */
    CONSTRAINT FK_Appointment_Patient
        FOREIGN KEY (PATIENT_ID) REFERENCES Tb_Patients (ID),
    CONSTRAINT FK_Appointment_Doctor
        FOREIGN KEY (DOCTOR_ID) REFERENCES Tb_Doctors (ID)
)


USE Hospital
GO

-- Lists all patients in alphabetical order
SELECT NAME FROM Tb_Patients
ORDER BY NAME ASC

/* Lists doctors in alphabetical order, joining the specialties table
   to display specialty names instead of IDs for better readability */
SELECT Tb_Specialties.NAME AS Specialty, Tb_Doctors.NAME AS Doctor
FROM Tb_Doctors
JOIN Tb_Specialties ON Tb_Doctors.SPECIALTY_ID = Tb_Specialties.ID
ORDER BY Tb_Doctors.NAME ASC

/* Lists all appointments with patient names, doctor names and specialty,
   ordered by date for better user understanding */
SELECT Tb_Patients.NAME AS Patient, Tb_Appointments.APPOINTMENT_DATE AS Date,
       Tb_Doctors.NAME AS Doctor, Tb_Specialties.NAME AS Specialty
FROM Tb_Appointments
JOIN Tb_Patients    ON Tb_Appointments.PATIENT_ID = Tb_Patients.ID
JOIN Tb_Doctors     ON Tb_Appointments.DOCTOR_ID  = Tb_Doctors.ID
JOIN Tb_Specialties ON Tb_Doctors.SPECIALTY_ID    = Tb_Specialties.ID
ORDER BY APPOINTMENT_DATE ASC

-- Lists each doctor with their total number of appointments,
-- ordered by appointment count in descending order
SELECT Tb_Doctors.NAME, COUNT(*) AS Total_Appointments
FROM Tb_Doctors
JOIN Tb_Appointments ON Tb_Doctors.ID = Tb_Appointments.DOCTOR_ID
GROUP BY Tb_Doctors.NAME
ORDER BY Total_Appointments DESC
